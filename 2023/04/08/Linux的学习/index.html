100以内奇数偶数分类

#!/bin/bash

echo "奇数  偶数"
sum=0
for i in {1..100}
do
    if [ $((i%2)) -eq 0 ]
    then
        echo "  $i"
    else
        echo -e "$i \c "
        if [ $i -le 10 ]
        then
            sum=$((sum+i))
        fi
    fi
done
echo "前10行数字的和为: $sum"
